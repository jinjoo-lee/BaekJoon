def func(n):
    a = 0
    for i in list(str(n)):
        a = a + int(i)
    return int(n) + a
a = [] # 생성자 담을 리스트
for i in range(1, 10001): # 생성자 찾기
    k = func(i)
    a.append(k)
for j in range(1, 10001): 
    if j in a: # 생성자면 pass
        pass
    else: # 셀프넘버인 경우 출력
        print(j)
